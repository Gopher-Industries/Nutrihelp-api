## Required SQL Migrations

```sql
-- MIGRATION 1: Create token_activity_logs table
CREATE TABLE token_activity_logs (
  id BIGINT PRIMARY KEY GENERATED ALWAYS AS IDENTITY,
  user_id BIGINT,
  event_type TEXT,          -- 'token_generated', 'token_verified', 'token_refreshed'
  token_type TEXT,          -- 'access', 'refresh'
  ip_address VARCHAR(45),
  user_agent TEXT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- MIGRATION 2: Create session_logs table
CREATE TABLE session_logs (
  id BIGINT PRIMARY KEY GENERATED ALWAYS AS IDENTITY,
  user_id BIGINT,
  event TEXT,               -- 'login', 'logout', 'session_expired'
  timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- MIGRATION 3: Create system_logs table
CREATE TABLE system_logs (
  id BIGINT PRIMARY KEY GENERATED ALWAYS AS IDENTITY,
  event TEXT,               -- 'server_start', 'server_stop'
  port INTEGER,
  timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
```
